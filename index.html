      Get Started
      
Install Lyra in 60 seconds

    

    

      
      
      
    

    

      

        
1
Download the zip
Download lyra-lang-2.0.0.zip

        
2
Extract to Downloads folder
Right-click → Extract All → C:\Users\YourName\Downloads

        
3
Open PowerShell and run
cd "C:\Users\YourName\Downloads\lyra-lang"
powershell -ExecutionPolicy Bypass -File ".\install.ps1"

        
4
Open a NEW terminal and verify
lyra version

      

      

        
⚠️ Troubleshooting

        
    
          
  • lyra not recognized: Close terminal completely, open a new one
    • 
          
  • Script blocked: Use powershell -ExecutionPolicy Bypass -File ".\install.ps1"
    • 
          
  • Python not found: Install from python.org — tick "Add Python to PATH"
    • 
          
  • Still shows old version: Run Remove-Item -Recurse -Force "$env:APPDATA\Lyra" then reinstall
    • 
          
  • Space in username errors: Always wrap paths in double quotes
    •

sentiment.lyr

          
note AI Sentiment Analyzer.
note Trains on examples, classifies new text.
use ml.

let reviews be an empty list.
add "brilliant product love it" to reviews.
add "absolutely outstanding quality" to reviews.
add "terrible waste of money awful" to reviews.
add "broke on first day horrible" to reviews.
add "decent nothing special" to reviews.

let labels be a list of
  "positive", "positive",
  "negative", "negative",
  "neutral".

let model be trainbayes with reviews, labels.

let tests be a list of
  "amazing fast delivery loved it",
  "stopped working after one week",
  "it arrived on time".

for each text in tests.
  let verdict be predictbayes with model, text.
  say verdict followed by "  -- "
    followed by text.
end for.

        

        

          
Output

          
positive  -- amazing fast delivery loved it
negative  -- stopped working after one week
neutral   -- it arrived on time

          
Trained from scratch. No external libraries.

security.lyr

          
note Password security and hashing tool.
use crypto.

say "Password Security Tool".
say "".

note Hash a password securely
let password be "mySecretPass99".
let hashed be hash with password.
say "Original   " followed by password.
say "SHA256     " followed by hashed.
say "".

note Check password strength
let passwords be a list of
  "abc",
  "password123",
  "Tr0ub4dor",
  "Zx9mK2qL8vN".

say "Strength Analysis".
for each pwd in passwords.
  let result be passwordstrength with pwd.
  let rating be result at "rating".
  let score be result at "score".
  say "  " followed by pwd
    followed by "  " followed by rating
    followed by "  score " followed by score.
end for.

say "Secure token  " followed by secure_token.

github_stats.lyr

          
note Live GitHub API client.
note Fetches real data from the internet.

ask "Enter a GitHub repo, e.g. torvalds slash linux."
  into repo.

let base be "https://api.github.com/repos/".
let url be base joined with repo.

say "Fetching live data...".

try.
  let response be get url.
  let data be parse response as json.

  let name be data at "full_name".
  let stars be data at "stargazers_count".
  let forks be data at "forks_count".
  let lang be data at "language".
  let issues be data at "open_issues_count".
  let desc be data at "description".

  say "".
  say "Repository  " followed by name.
  say "Language    " followed by lang.
  say "Stars       " followed by stars.
  say "Forks       " followed by forks.
  say "Open issues " followed by issues.
  say "Description " followed by desc.

catch err.
  say "Could not fetch. Check repo name.".
end try.

        

        

          
Output — live result

Stars        185000
Forks        54000

Description  Linux kernel source tree

classes.lyr

          
note Object-oriented programming in plain English.
note Classes, inheritance, methods.

define class BankAccount.
  it has owner, balance.

  define create taking owner, opening_balance.
    set its owner to owner.
    set its balance to opening_balance.
  end define.

  define deposit taking amount.
    add amount to its balance.
    say "Deposited " followed by amount.
  end define.

  define withdraw taking amount.
    if amount is greater than its balance.
      say "Insufficient funds.".
      return.
    end if.
    subtract amount from its balance.
    say "Withdrew " followed by amount.
  end define.

  define show_balance.
    say its owner followed by
      "  balance  " followed by its balance.
  end define.
end class.

let acc be a new BankAccount
  with "Alice", 1000.
call deposit on acc with 500.
call withdraw on acc with 200.
call withdraw on acc with 5000.
call show_balance on acc.

        

        

          
Output

          
Deposited 500
Withdrew 200
Insufficient funds.
Alice  balance  1300
